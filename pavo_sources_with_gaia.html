# Kiman et al 2024
a = -0.143
b = 1.22
c = 0.92
d = -0.017

color_index = gaia_df["phot_bp_mean_mag"] - gaia_df["phot_rp_mean_mag"]
metallicity = gaia_df["mh_gspphot"]
prod = (a * color_index**2 + b * color_index + c) * (1 + d * metallicity)
po = (prod - 0.4 * gaia_df["phot_g_mean_mag"]) / 2
angular_diameter = 10**po

physical_radii = (107.47 * angular_diameter/gaia_df["parallax"])
